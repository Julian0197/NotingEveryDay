### 1.合并两个有序链表

当你需要创造一条新链表的时候，可以使用虚拟头结点简化边界情况的处理。

~~~js
var mergeTwoLists = function(list1, list2) {
    // 创建虚拟头结点
    let dummy = new ListNode(-1)
    // 保存头结点用于返回，p不断移动
    let p = dummy
    // 有一个为空就退出循环
    while (list1 !== null && list2 !== null) {
        if (list1.val <= list2.val) {
            p.next = list1
            list1 = list1.next
        } else {
            p.next = list2
            list2 = list2.next
        }
        // 向后移动指针
        p = p.next
    }
    // 退出循环可能list1或list2有一个没遍历完
    p.next = list1 === null ? list2 : list1
    return dummy.next
};
~~~

### 2.分隔链表

链表按照给定值分成大于它的和小于它的链表，再拼接两个链表

~~~js
var partition = function(head, x) {
    // 新建两个链表
    let dummy1 = new ListNode(-1)
    let dummy2 = new ListNode(-1)
    let p1 = dummy1, p2 = dummy2
    // p负责遍历原链表
    let p = head
    while (p !== null) {
        if (p.val >= x) {
            p2.next = p
            p2 = p2.next
        } else {
            p1.next = p
            p1 = p1.next
        }
        // 断开原链表中每个节点的next
        let temp = p.next
        p.next = null
        p = temp
    }
    // 拼接两个链表
    p1.next = dummy2.next
    return dummy1.next
};
~~~

### 3.合并k个升序链表
建立优先级队列（小顶堆：父元素始终小于孩子元素，堆头元素最小）
+ 构建小顶堆，依次把各个链表头元素加入堆
+ 弹出堆顶元素，拼接到新链表；将堆顶元素所在链表的下一个元素推入堆内重新排序
+ 将所有元素推入堆后，合并工作完成

时间复杂度：优先队列最多元素个数为k，一次pop和一次insert的时间复杂度为O(logk)；所有链表元素都会被加入和弹出，所以总的时间复杂度是O(nlogk)
~~~js
// 小顶堆实现（父亲元素始终小于两个孩子）
class MiniHeap {
    constructor() {
        this.heap = []
    }

    // 交换元素
    swap(i1, i2) {
        [this.heap[i1], this.heap[i2]] = [this.heap[i2], this.heap[i1]]
    }
    // 获得父节点
    getParentIndex(i) {
        return (i - 1) >> 1
    }
    // 获得左孩子
    getLeftIndex(i) {
        return 2 * i + 1
    }
    // 获得右孩子
    getRightIndex(i) {
        return 2 * i + 2
    }
    // 上浮
    shiftUp(i) {
        if (i === 0) return 
        const parentIndex = this.getParentIndex(i)
        // 当前元素小于父元素，需要上浮
        if (this.heap[parentIndex] && this.heap[i].val < this.heap[parentIndex].val) {
            this.swap(parentIndex, i)
            // 继续上浮
            this.shiftUp(parentIndex)
        } 
    }
    // 下沉
    shiftDown(i) {
        const leftIndex = this.getLeftIndex(i)
        const rightIndex = this.getRightIndex(i)
        // 当前元素大于子元素，需要下沉
        if (this.heap[leftIndex] && this.heap[i].val > this.heap[leftIndex].val) {
            this.swap(leftIndex, i)
            this.shiftDown(leftIndex)
        }
        if (this.heap[rightIndex] && this.heap[i].val > this.heap[rightIndex].val) {
            this.swap(rightIndex, i)
            this.shiftDown(rightIndex)
        }
    }
    // 插入
    insert(value) {
        this.heap.push(value)
        this.shiftUp(this.heap.length - 1)
    }
    // 删除堆顶元素
    pop() {
        if (this.size() === 1) return this.heap.shift()
        const top = this.heap[0]
        // 最后一个元素放到堆头，再下沉重新排序
        this.heap[0] = this.heap.pop()
        this.shiftDown(0)
        return top
    }
    // 获取堆顶
    peak() {
        return this.heap[0]
    }
    // 堆的大小
    size() {
        return this.heap.length
    }
}

var mergeKLists = function(lists) {
    const dummy = new ListNode(-1)
    let p = dummy
    const heap = new MiniHeap()
    // 将所有链表头元素添加到小顶堆
    lists.forEach(i => {
        if(i) heap.insert(i)
    })
    // 不断向小顶堆中插入元素,并取出堆顶元素添加到新链表中
    while (heap.size()) {
        const n = heap.pop()
        p.next = n
        p = p.next
        if (n.next) heap.insert(n.next)
    }
    return dummy.next
};
~~~
