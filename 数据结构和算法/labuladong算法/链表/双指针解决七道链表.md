### 1.合并两个有序链表

当你需要创造一条新链表的时候，可以使用虚拟头结点简化边界情况的处理。

~~~js
var mergeTwoLists = function(list1, list2) {
    // 创建虚拟头结点
    let dummy = new ListNode(-1)
    // 保存头结点用于返回，p不断移动
    let p = dummy
    // 有一个为空就退出循环
    while (list1 !== null && list2 !== null) {
        if (list1.val <= list2.val) {
            p.next = list1
            list1 = list1.next
        } else {
            p.next = list2
            list2 = list2.next
        }
        // 向后移动指针
        p = p.next
    }
    // 退出循环可能list1或list2有一个没遍历完
    p.next = list1 === null ? list2 : list1
    return dummy.next
};
~~~

### 2.分隔链表

链表按照给定值分成大于它的和小于它的链表，再拼接两个链表

~~~js
var partition = function(head, x) {
    // 新建两个链表
    let dummy1 = new ListNode(-1)
    let dummy2 = new ListNode(-1)
    let p1 = dummy1, p2 = dummy2
    // p负责遍历原链表
    let p = head
    while (p !== null) {
        if (p.val >= x) {
            p2.next = p
            p2 = p2.next
        } else {
            p1.next = p
            p1 = p1.next
        }
        // 断开原链表中每个节点的next
        let temp = p.next
        p.next = null
        p = temp
    }
    // 拼接两个链表
    p1.next = dummy2.next
    return dummy1.next
};
~~~
